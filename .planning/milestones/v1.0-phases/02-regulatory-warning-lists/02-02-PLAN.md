Create src/components/entity/WarningBadge.tsx as a new file. Model directly after SanctionBadge.tsx — use the Badge primitive, no external dependencies.

```typescript
import Badge from '@/components/ui/Badge'

interface WarningBadgeProps {
  source: string         // 'fca' | 'finma' | 'sfc' | 'mas' | 'dfsa' | 'sca' | 'cma'
  sourceName: string     // 'FCA (UK)' | 'FINMA (Switzerland)' | etc.
  jurisdiction: string   // 'UK' | 'CH' | 'HK' | 'SG' | 'AE-DU' | 'AE' | 'OM'
  size?: 'sm' | 'md'
}

// Label map: source → display label (middle dot U+00B7)
const LABEL: Record<string, string> = {
  fca:   'FCA \u00B7 UK',
  finma: 'FINMA \u00B7 CH',
  sfc:   'SFC \u00B7 HK',
  mas:   'MAS \u00B7 SG',
  dfsa:  'DFSA \u00B7 Dubai',
  sca:   'SCA \u00B7 UAE',
  cma:   'CMA \u00B7 Oman',
}

export default function WarningBadge({ source, sourceName, jurisdiction: _jurisdiction, size = 'md' }: WarningBadgeProps) {
  const label = LABEL[source] ?? `${source.toUpperCase()} \u00B7 Warning`
  const tooltip = `${sourceName} \u2014 Regulatory Warning List`

  return (
    <span title={tooltip}>
      <Badge
        label={label}
        color="var(--accent-amber)"
        background="rgba(245, 158, 11, 0.12)"
        size={size}
        // No className — WarningBadge has NO glow/pulse animation (per D-02 and UI-SPEC)
      />
    </span>
  )
}
```

Key implementation notes:
- `\u00B7` is the middle dot character (·) — the only approved separator per UI-SPEC
- `\u2014` is the em dash (—) used in tooltip
- No `className` on Badge — no glow animation, this is intentional
- The outer `<span title={tooltip}>` provides the native browser tooltip (no JS library)
- `jurisdiction` prop is kept in interface for forward-compatibility but unused in label rendering (label table is source-driven)

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| DB → server component | SQL rows flow from regulatory_warnings into page render — executor is a trusted server component |
| entity_name → badge label | entity_name from DB is displayed in tooltip title attribute — XSS surface |
| hit.source → LABEL lookup | source key controls label displayed — should be a known-key fallback |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-02-01 | Information Disclosure | WarningBadge tooltip title= | mitigate | `title` attribute is plain text — React renders it escaped by default. No dangerouslySetInnerHTML. XSS risk is negligible |
| T-02-02-02 | Tampering | LABEL[source] lookup | mitigate | Unknown source key falls back to `source.toUpperCase() + ' · Warning'` — no crash, no code injection, no undefined access |
| T-02-02-03 | Denial of Service | getWarningHits() on every entity page load | accept | Query is indexed (GIN trigram index on normalized_name). For empty/no-match cases returns [] with minimal DB work. No caching added in Phase 2 — acceptable for current traffic volume |
| T-02-02-04 | Information Disclosure | WarningBadge visible to unauthenticated users | accept | Intentional — WarningBadge is F1 (always free per D-02 and UI-SPEC). Warning list membership is public information published by regulators |
| T-02-02-05 | Spoofing | Malicious entity_name in DB displayed in badge | mitigate | entity_name only appears in tooltip title= attribute (escaped by browser); badge label comes from the hardcoded LABEL map keyed by source, not from entity_name. LABEL map is closed/enumerated — no injection path |
</threat_model>
