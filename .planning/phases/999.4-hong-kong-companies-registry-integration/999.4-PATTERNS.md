```typescript
const CKAN_PACKAGE_URL = 'https://data.gov.hk/en-data/api/3/action/package_show?id=hk-cr-crdata-list-newly-registered-companies-2526'

async function fetchCSVResourceUrls(): Promise<string[]> {
  const response = await fetch(CKAN_PACKAGE_URL, {
    headers: { 'User-Agent': 'ETI-Bot/1.0' },
    signal: AbortSignal.timeout(30000),
  })
  const data = await response.json()
  if (!data.success) throw new Error('CKAN package fetch failed')
  
  // Extract CSV resource URLs from package
  return data.result.resources
    .filter((r: any) => r.format === 'CSV' || r.format === 'XLSX')
    .map((r: any) => r.url)
}
```

**CSV parse + import pattern** (from gleif-golden-copy.ts syncLeiGoldenCopy):
```typescript
interface HKCRRow {
  Company_Number: string
  Company_Name: string
  Company_Name_Chinese?: string
  Company_Type?: string
  Company_Status?: string
  Date_of_Incorporation?: string
}

export async function syncHKCRFull(): Promise<{ inserted: number; errors: number }> {
  const csvUrls = await fetchCSVResourceUrls()
  let totalInserted = 0
  let totalErrors = 0

  for (const url of csvUrls) {
    try {
      const response = await fetch(url, { signal: AbortSignal.timeout(60000) })
      const text = await response.text()
      
      const parser = csvParse(text, { columns: true, skip_empty_lines: true })
      const records: HKCRRow[] = []
      
      for await (const record of parser) {
        records.push({
          Company_Number: record['Company Number'] || record['company_number'],
          Company_Name: record['Company Name'] || record['company_name'],
          Company_Name_Chinese: record['Company Name (Chinese)'],
          Company_Type: record['Company Type'],
          Company_Status: record['Company Status'],
          Date_of_Incorporation: record['Date of Incorporation'],
        })
      }
      
      // Batch insert
      const inserted = await batchInsertHKCR(records)
      totalInserted += inserted
    } catch (err) {
      totalErrors++
      console.error(`Failed to process ${url}:`, err)
    }
  }
  
  return { inserted: totalInserted, errors: totalErrors }
}

async function batchInsertHKCR(records: HKCRRow[]): Promise<number> {
  if (records.length === 0) return 0
  
  const values = records.map((r) => [
    r.Company_Number,
    r.Company_Name,
    r.Company_Name_Chinese || null,
    r.Company_Type || null,
    r.Company_Status || null,
    r.Date_of_Incorporation ? new Date(r.Date_of_Incorporation) : null,
  ])
  
  const result = await db.query(`
    INSERT INTO hkcr_cache (company_number, company_name, company_name_chinese, company_type, company_status, date_of_incorporation)
    VALUES ($1, $2, $3, $4, $5, $6)
    ON CONFLICT (company_number) DO UPDATE SET
      company_name = EXCLUDED.company_name,
      company_status = EXCLUDED.company_status,
      fetched_at = NOW()
  `, values.length === 1 ? values[0] : values.flat())
  
  return result.rowCount || 0
}
```

**Local cache query pattern** (for Tier 2):
```typescript
export interface HKCRCacheResult {
  company_number: string
  company_name: string
  company_name_chinese?: string
  company_type?: string
  company_status?: string
  date_of_incorporation?: Date
}

export async function searchHKCRCache(query: string, limit = 5): Promise<HKCRCacheResult[]> {
  if (!query || query.trim().length < 2) return []
  
  const normalizedQuery = query.trim()
  
  // Check if query is a HK number (7-digit)
  if (mightBeHKNumber(normalizedQuery)) {
    const { rows } = await db.query<HKCRCacheResult>(
      `SELECT company_number, company_name, company_name_chinese, company_type, company_status, date_of_incorporation
       FROM hkcr_cache
       WHERE company_number = $1
       LIMIT $2`,
      [normalizedQuery, limit]
    )
    return rows
  }
  
  // Name search with pg_trgm
  const { rows } = await db.query<HKCRCacheResult>(
    `SELECT company_number, company_name, company_name_chinese, company_type, company_status, date_of_incorporation
       FROM hkcr_cache
       WHERE company_name ILIKE $1 OR company_name_chinese ILIKE $1
       ORDER BY company_name <-> $2
       LIMIT $3`,
    [`%${normalizedQuery}%`, normalizedQuery, limit]
  )
  
  return rows
}

/** Convert HK cache result to SearchResult */
export function hkcrToSearchResult(c: HKCRCacheResult) {
  const authenticityScore = 20  // Entity existence: official registry data (20/25)
  return {
    id:                 `hkcr:${c.company_number}`,
    name:               c.company_name,
    type:               'company' as const,
    country:            'Hong Kong',
    jurisdictionFlag:   '🇭🇰',
    sanctionStatus:     'unknown' as const,
    authenticityScore,
    riskLevel:          riskLevel(authenticityScore, 'unknown'),
    registrationNumber: c.company_number,
    slug:               `hk-${c.company_number}`,
  }
}
```

## Anti-Patterns to Avoid

### Anti-Pattern 1: Calling data.gov.hk datastore_search

**Why avoid:** data.gov.hk CKAN API does NOT have a searchable HK company database.

**Evidence:** RESEARCH.md confirms — only weekly CSV files, no datastore.

**Alternative:** Batch download CSV files and query local cache.

### Anti-Pattern 2: Using OpenCorporates as primary source

**Why avoid:** OpenCorporates **无免费层**，超出项目预算。

**Alternative:** 使用免费的 data.gov.hk CSV 文件 + 本地缓存。

### Anti-Pattern 3: Real-time API query for HK entities

**Why avoid:** HK 无免费实时搜索 API。

**Alternative:** 批量导入到本地缓存表，Tier 2 查询本地数据库。
