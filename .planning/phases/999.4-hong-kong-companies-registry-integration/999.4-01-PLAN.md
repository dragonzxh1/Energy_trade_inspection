       ```typescript
       export async function searchHKCR(query: string, limit = 5): Promise<OCCompany[]> {
         if (!query || query.trim().length < 2) return []
         const ocResults = await searchOpenCorporates(query, limit * 2).catch(() => [])
         return ocResults
           .filter((c) => c.jurisdiction_code === HK_JURISDICTION || c.jurisdiction_code.startsWith('hk_'))
           .slice(0, limit)
       }
       ```
