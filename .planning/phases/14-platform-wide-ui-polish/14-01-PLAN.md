          display: 'flex', flexDirection: 'column',
          alignItems: 'center', justifyContent: 'center',
          minHeight: '400px', gap: '16px',
        }}>

            background: 'rgba(0,0,0,0.35)',

          }}>

              width: '0%', transition: 'width 1.4s ease',
            }} />
          </div>
